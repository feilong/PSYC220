from shiny import App, render, ui, reactive
import numpy as np
import matplotlib.pyplot as plt

# 1. Define the User Interface
app_ui = ui.page_fluid(
    ui.layout_sidebar(
        ui.sidebar(
            ui.input_select(
                "dist", "Base Distribution Shape",
                {"norm": "Normal (Symmetric, Bell)", 
                 "unif": "Uniform (Flat, Rectangular)", 
                 "exp": "Exponential (Right-Skewed)",
                 "bimodal": "Bimodal (Two Peaks)",
                 "beta": "Beta (Left-Skewed)"}
            ),
            # These inputs are now always visible and apply to ALL distributions
            ui.input_numeric("pop_mu", "Set Population Mean (μ):", value=0.0, step=1.0),
            ui.input_numeric("pop_var", "Set Population Variance (σ²):", value=1.0, min=0.1, step=0.5),
            ui.hr(),
            ui.input_action_button("draw_1", "Add 1 Score", class_="btn-primary"),
            ui.input_action_button("draw_10", "Add 10 Scores", class_="btn-success"),
            ui.input_action_button("reset", "Start Over", class_="btn-danger"),
            ui.markdown("Compare calculations using the **Population Mean (μ)** vs. the **Sample Mean (M)**."),
            width=300
        ),
        ui.output_plot("main_plot", height="800px")
    )
)

# 2. Define the Server Logic
def server(input, output, session):
    sample_data = reactive.Value([])

    # Mathematical properties of the raw, unscaled base distributions
    def get_base_params(d):
        if d == "norm":      return 0.0, 1.0
        elif d == "unif":    return 0.5, np.sqrt(1.0 / 12.0)
        elif d == "exp":     return 1.0, 1.0
        elif d == "bimodal": return 0.0, np.sqrt(4.25)
        elif d == "beta":    return 0.8, np.sqrt(16.0 / 1100.0)

    # Generate raw scores, then standardize and scale them to match user inputs
    def draw_scaled_scores(n_scores):
        d = input.dist()
        target_mu = float(input.pop_mu())
        target_sigma = np.sqrt(float(input.pop_var()))
        base_mu, base_sigma = get_base_params(d)
        
        # 1. Draw from raw distribution
        if d == "norm":      raw = np.random.standard_normal(n_scores)
        elif d == "unif":    raw = np.random.uniform(size=n_scores)
        elif d == "exp":     raw = np.random.exponential(size=n_scores)
        elif d == "bimodal": 
            centers = np.where(np.random.rand(n_scores) < 0.5, -2, 2)
            raw = np.random.normal(centers, 0.5)
        elif d == "beta":    raw = np.random.beta(8, 2, size=n_scores)
            
        # 2. Scale and shift to match user's mu and sigma
        scaled_scores = (raw - base_mu) * (target_sigma / base_sigma) + target_mu
        return scaled_scores.tolist()

    @reactive.Effect
    @reactive.event(input.draw_1)
    def add_one():
        sample_data.set(sample_data() + draw_scaled_scores(1))

    @reactive.Effect
    @reactive.event(input.draw_10)
    def add_ten():
        sample_data.set(sample_data() + draw_scaled_scores(10))

    @reactive.Effect
    @reactive.event(input.reset, input.dist, input.pop_mu, input.pop_var) 
    def clear_data():
        sample_data.set([])

    # 3. Render the Plots
    @output
    @render.plot
    def main_plot():
        data = sample_data()
        n = len(data)
        d = input.dist()
        
        target_mu = float(input.pop_mu())
        true_var = float(input.pop_var())
        target_sigma = np.sqrt(true_var)
        base_mu, base_sigma = get_base_params(d)
        
        fig, (ax1, ax2, ax3) = plt.subplots(3, 1, figsize=(10, 10))
        fig.tight_layout(pad=5.0)
        
        # --- TOP PLOT: Population Distribution ---
        # Dynamically frame the x-axis for ALL distributions based on sigma
        x_vals = np.linspace(target_mu - 4*target_sigma, target_mu + 4*target_sigma, 1000)
        
        # To draw the scaled PDF, we reverse-map the x-axis back to the base distribution
        scale_factor = target_sigma / base_sigma
        x_base = (x_vals - target_mu) / scale_factor + base_mu
        
        if d == "norm":   
            y_base = (1 / np.sqrt(2 * np.pi)) * np.exp(-0.5 * x_base**2)
        elif d == "unif": 
            y_base = np.where((x_base >= 0) & (x_base <= 1), 1, 0)
        elif d == "exp":  
            y_base = np.where(x_base >= 0, np.exp(-x_base), 0)
        elif d == "bimodal":
            y_base = (1 / np.sqrt(2 * np.pi)) * (np.exp(-2 * (x_base + 2)**2) + np.exp(-2 * (x_base - 2)**2))
        elif d == "beta":
            y_base = np.where((x_base >= 0) & (x_base <= 1), 72 * (x_base**7) * (1 - x_base), 0)
            
        # Scale the height of the PDF so the area underneath remains 1
        y_vals = y_base / scale_factor
            
        ax1.plot(x_vals, y_vals, color='gray', linewidth=2)
        ax1.fill_between(x_vals, y_vals, alpha=0.2, color='gray')
        ax1.axvline(target_mu, color='black', linestyle='--', label=f'Population mean $μ$ ({target_mu:.2f})')
        ax1.set_xlim(target_mu - 4*target_sigma, target_mu + 4*target_sigma)
        
        if n > 0:
            ax1.scatter(data, np.zeros(n), color='dodgerblue', zorder=5, label='Sample Points', alpha=0.6)
            ax1.axvline(np.mean(data), color='red', linestyle='-', label=f'Sample mean $M$ ({np.mean(data):.2f})')
            
        ax1.set_title("Population Distribution & Sample Scores")
        ax1.legend(loc='upper right')
        
        # Base state before any data is drawn for bottom two plots
        if n == 0:
            for ax in [ax2, ax3]:
                ax.set_xlim(0, 10)
                ax.set_ylim(0, max(5, true_var * 2))
                ax.text(5, max(5, true_var * 2) / 2, "Click 'Add 1 Score' to begin", ha='center', va='center', fontsize=12)
            ax2.set_title("Sum of Squares Growth")
            ax3.set_title("Variance Estimates")
            return fig
            
        # --- Calculate Histories ---
        n_values = np.arange(1, n + 1)
        ss_mu_vals, ss_xbar_vals, diff_ss_vals = [], [], []
        var_mu_vals, var_xbar_vals, diff_var_vals = [], [], []
        
        for i in range(1, n + 1):
            subset = np.array(data[:i])
            xbar = np.mean(subset)
            
            # Sum of Squares
            ss_mu = np.sum((subset - target_mu)**2)
            ss_xbar = np.sum((subset - xbar)**2)
            diff_ss = i * (target_mu - xbar)**2 
            
            ss_mu_vals.append(ss_mu)
            ss_xbar_vals.append(ss_xbar)
            diff_ss_vals.append(diff_ss)
            
            # Variance
            var_mu = ss_mu / i
            var_xbar = ss_xbar / (i - 1) if i > 1 else np.nan
            diff_var = (target_mu - xbar)**2
            
            var_mu_vals.append(var_mu)
            var_xbar_vals.append(var_xbar)
            diff_var_vals.append(diff_var)
                
        # --- MIDDLE PLOT: Sum of Squares ---
        ax2.plot(n_values, ss_mu_vals, color='purple', marker='o', markersize=4, label='SS using $\\sum (x_i - μ)^2$')
        ax2.plot(n_values, ss_xbar_vals, color='darkorange', marker='s', markersize=4, label='SS using $\\sum (x_i - M)^2$')
        ax2.plot(n_values, diff_ss_vals, color='seagreen', marker='^', markersize=4, label='$n(μ - M)^2$')
        
        ax2.set_title("Sum of Squares (SS) vs. Sample Size")
        ax2.set_xlabel("Sample Size (n)")
        ax2.set_ylabel("Sum of Squares")
        ax2.set_xlim(0, max(10, n + 1))
        ax2.legend(loc='upper left')
        
        # --- BOTTOM PLOT: Variance ---
        ax3.axhline(true_var, color='black', linestyle='--', linewidth=2, label=f'True σ² ({true_var:.3f})')
        ax3.plot(n_values, var_mu_vals, color='purple', marker='o', markersize=4, alpha=0.7, label='$s^2$ using $\\frac{\\sum (x_i - \\mu)^2}{n}$')
        if n > 1:
            ax3.plot(n_values, var_xbar_vals, color='darkorange', marker='s', markersize=4, alpha=0.7, label='$s^2$ using $\\frac{\\sum (x_i - M)^2}{n-1}$')
        ax3.plot(n_values, diff_ss_vals, color='seagreen', marker='^', markersize=4, alpha=0.7, label='$n(μ - M)^2$')
            
        ax3.set_title("Variance vs. Sample Size")
        ax3.set_xlabel("Sample Size (n)")
        ax3.set_ylabel("Variance")
        ax3.set_xlim(0, max(10, n + 1))
        
        # Dynamic y-axis for variance
        max_var = max(max(var_mu_vals), np.nanmax(var_xbar_vals) if n > 1 else 0, max(diff_ss_vals), true_var)
        ax3.set_ylim(0, max_var * 1.5)
        ax3.legend(loc='upper right')
        
        return fig

app = App(app_ui, server)
