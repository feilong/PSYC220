from shiny import App, render, ui, reactive
import numpy as np
import matplotlib.pyplot as plt

# 1. Define the User Interface
app_ui = ui.page_fluid(
    ui.layout_sidebar(
        ui.sidebar(
            ui.input_select(
                "dist", "Base Distribution Shape",
                {"norm": "Normal (Symmetric, Bell)", 
                 "unif": "Uniform (Flat, Rectangular)",
                 "exp": "Exponential (Right-Skewed)",
                 "bimodal": "Bimodal (Two Peaks)",
                 "beta": "Beta (Left-Skewed)"}
            ),
            ui.input_numeric("pop_mu", "Set Population Mean (μ):", value=0.0, step=1.0),
            ui.input_numeric("pop_var", "Set Population Variance (σ²):", value=1.0, min=0.1, step=0.5),
            ui.hr(),
            ui.input_numeric("n1", "Group 1 Size (n₁):", value=30, min=2, step=1),
            ui.input_numeric("n2", "Group 2 Size (n₂):", value=30, min=2, step=1),
            ui.hr(),
            ui.input_action_button("draw_1", "Draw 1 Sample", class_="btn-primary"),
            ui.input_action_button("draw_10", "Draw 10 Samples", class_="btn-success"),
            ui.input_action_button("draw_100", "Draw 100 Samples", class_="btn-warning"),
            ui.input_action_button("reset", "Clear History", class_="btn-danger"),
            ui.markdown("Compare the Sum of Squares (SS) and Mean Squares (MS) within each group versus between the sample means."),
            width=300
        ),
        ui.output_plot("main_plot", height="900px")
    )
)

# 2. Define the Server Logic
def server(input, output, session):
    history = reactive.Value({
        'm_grand': [], 'm_diff': [], 
        'ss1': [], 'ss2': [], 'diff_ss': [], 'ss_grand': [],
        'var1': [], 'var2': [], 'diff_var': [], 'var_grand': []
    })

    def get_base_params(d):
        if d == "norm":      return 0.0, 1.0
        elif d == "unif":    return 0.5, np.sqrt(1.0 / 12.0)
        elif d == "exp":     return 1.0, 1.0
        elif d == "bimodal": return 0.0, np.sqrt(4.25)
        elif d == "beta":    return 0.8, np.sqrt(16.0 / 1100.0)

    def draw_k_samples(k):
        d = input.dist()
        target_mu = float(input.pop_mu())
        target_sigma = np.sqrt(float(input.pop_var()))
        base_mu, base_sigma = get_base_params(d)
        n1 = int(input.n1())
        n2 = int(input.n2())
        n_total = n1 + n2
        
        def get_raw(n):
            size = (k, n)
            if d == "norm":      return np.random.standard_normal(size)
            elif d == "unif":    return np.random.uniform(size=size)
            elif d == "exp":     return np.random.exponential(size=size)
            elif d == "bimodal": 
                centers = np.where(np.random.rand(k, n) < 0.5, -2, 2)
                return np.random.normal(centers, 0.5)
            elif d == "beta":    return np.random.beta(8, 2, size=size)

        # Process Group 1
        raw1 = get_raw(n1)
        scaled1 = (raw1 - base_mu) * (target_sigma / base_sigma) + target_mu
        m1 = np.mean(scaled1, axis=1)
        ss1 = np.sum((scaled1 - m1[:, None])**2, axis=1)
        var1 = ss1 / (n1 - 1) if n1 > 1 else np.zeros(k) * np.nan
        
        # Process Group 2
        raw2 = get_raw(n2)
        scaled2 = (raw2 - base_mu) * (target_sigma / base_sigma) + target_mu
        m2 = np.mean(scaled2, axis=1)
        ss2 = np.sum((scaled2 - m2[:, None])**2, axis=1)
        var2 = ss2 / (n2 - 1) if n2 > 1 else np.zeros(k) * np.nan
        
        # Process Grand Mean and Mean Difference
        m_grand = (n1 * m1 + n2 * m2) / n_total
        m_diff = m1 - m2
        
        # Process Between-Groups (Difference)
        diff_ss = (n1 * n2 / n_total) * (m1 - m2)**2
        diff_var = diff_ss / 1.0 
        
        # Process Grand Mean vs Population Mean
        ss_grand = n_total * (m_grand - target_mu)**2
        var_grand = ss_grand / 1.0
        
        curr = history.get().copy()
        curr['m_grand'].extend(m_grand.tolist())
        curr['m_diff'].extend(m_diff.tolist())
        curr['ss1'].extend(ss1.tolist())
        curr['ss2'].extend(ss2.tolist())
        curr['diff_ss'].extend(diff_ss.tolist())
        curr['ss_grand'].extend(ss_grand.tolist())
        curr['var1'].extend(var1.tolist())
        curr['var2'].extend(var2.tolist())
        curr['diff_var'].extend(diff_var.tolist())
        curr['var_grand'].extend(var_grand.tolist())
        history.set(curr)

    @reactive.Effect
    @reactive.event(input.draw_1)
    def add_1(): draw_k_samples(1)

    @reactive.Effect
    @reactive.event(input.draw_10)
    def add_10(): draw_k_samples(10)

    @reactive.Effect
    @reactive.event(input.draw_100)
    def add_100(): draw_k_samples(100)

    @reactive.Effect
    @reactive.event(input.reset, input.dist, input.pop_mu, input.pop_var, input.n1, input.n2) 
    def clear_data():
        history.set({'m_grand': [], 'm_diff': [], 'ss1': [], 'ss2': [], 'diff_ss': [], 'ss_grand': [], 
                     'var1': [], 'var2': [], 'diff_var': [], 'var_grand': []})

    # 3. Render the Plots
    @output
    @render.plot
    def main_plot():
        curr = history.get()
        num_samples = len(curr['m_grand'])
        
        d = input.dist()
        target_mu = float(input.pop_mu())
        true_var = float(input.pop_var())
        target_sigma = np.sqrt(true_var)
        base_mu, base_sigma = get_base_params(d)
        n1 = int(input.n1())
        n2 = int(input.n2())
        
        fig = plt.figure(figsize=(16, 10))
        gs = fig.add_gridspec(2, 6)
        
        ax_pop = fig.add_subplot(gs[0, 0:2])
        ax_grand = fig.add_subplot(gs[0, 2:4])
        ax_diff = fig.add_subplot(gs[0, 4:6])
        ax_ss = fig.add_subplot(gs[1, 0:3])
        ax_ms = fig.add_subplot(gs[1, 3:6])
        
        fig.tight_layout(pad=5.0, w_pad=2.0, h_pad=4.0)
        
        # --- TOP LEFT: Population Distribution (Always Drawn) ---
        x_vals = np.linspace(target_mu - 4*target_sigma, target_mu + 4*target_sigma, 1000)
        scale_factor = target_sigma / base_sigma
        x_base = (x_vals - target_mu) / scale_factor + base_mu
        
        if d == "norm":   
            y_base = (1 / np.sqrt(2 * np.pi)) * np.exp(-0.5 * x_base**2)
        elif d == "unif": 
            y_base = np.where((x_base >= 0) & (x_base <= 1), 1, 0)
        elif d == "exp":  
            y_base = np.where(x_base >= 0, np.exp(-x_base), 0)
        elif d == "bimodal":
            y_base = (1 / np.sqrt(2 * np.pi)) * (np.exp(-2 * (x_base + 2)**2) + np.exp(-2 * (x_base - 2)**2))
        elif d == "beta":
            y_base = np.where((x_base >= 0) & (x_base <= 1), 72 * (x_base**7) * (1 - x_base), 0)
            
        y_vals = y_base / scale_factor
            
        ax_pop.plot(x_vals, y_vals, color='gray', linewidth=1)
        ax_pop.fill_between(x_vals, y_vals, alpha=0.2, color='gray')
        ax_pop.axvline(target_mu, color='black', linestyle='--', label=f'True μ ({target_mu:.2f})')
        ax_pop.set_xlim(target_mu - 4*target_sigma, target_mu + 4*target_sigma)
        ax_pop.set_title("Population Distribution")
        ax_pop.set_xlabel("Score Value")
        ax_pop.legend(loc='upper right')
        
        if num_samples == 0:
            for ax, title in zip([ax_grand, ax_diff, ax_ss, ax_ms], 
                                 ["Sampling Dist. of Grand Mean", "Sampling Dist. of Mean Diff (M₁ - M₂)", 
                                  "Sampling Dist. of Sum of Squares (SS)", "Sampling Dist. of Mean Squares (MS)"]):
                ax.set_xlim(0, 10)
                ax.set_ylim(0, 5)
                ax.text(5, 2.5, "Click 'Draw Samples' to begin", ha='center', va='center', fontsize=11)
                ax.set_title(title)
            return fig

        # --- TOP MIDDLE: Sampling Distribution of Grand Mean ---
        ax_grand.hist(curr['m_grand'], bins=20, density=True, color='dodgerblue', edgecolor='black', alpha=0.6)
        
        std_err_grand = target_sigma / np.sqrt(n1 + n2)
        x_g = np.linspace(target_mu - 4*std_err_grand, target_mu + 4*std_err_grand, 200)
        y_g = (1 / (std_err_grand * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x_g - target_mu) / std_err_grand)**2)
        
        ax_grand.plot(x_g, y_g, color='darkblue', linewidth=1, label='Theo. N(μ, σ²/(n₁+n₂))')
        ax_grand.axvline(target_mu, color='black', linestyle='--', linewidth=1, label=f'True μ ({target_mu:.2f})')
        ax_grand.axvline(np.mean(curr['m_grand']), color='blue', linestyle='-', linewidth=1, label=f'Mean of $M_{{grand}}$ ({np.mean(curr["m_grand"]):.2f})')
        
        ax_grand.set_title(f"Grand Mean (Total Samples: {num_samples})")
        ax_grand.set_xlabel("Grand Mean ($M_{grand}$)")
        ax_grand.legend(loc='upper right')

        # --- TOP RIGHT: Sampling Distribution of Mean Difference ---
        # Updated colors to visually link the "Mean Difference" with "Between Groups"
        ax_diff.hist(curr['m_diff'], bins=20, density=True, color='seagreen', edgecolor='black', alpha=0.6)
        
        std_err_diff = np.sqrt((true_var / n1) + (true_var / n2))
        x_d = np.linspace(0 - 4*std_err_diff, 0 + 4*std_err_diff, 200)
        y_d = (1 / (std_err_diff * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x_d - 0) / std_err_diff)**2)
        
        ax_diff.plot(x_d, y_d, color='darkgreen', linewidth=1, label='Theo. N(0, σ²/n₁ + σ²/n₂)')
        ax_diff.axvline(0, color='black', linestyle='--', linewidth=1, label='True Diff (0)')
        ax_diff.axvline(np.mean(curr['m_diff']), color='forestgreen', linestyle='-', linewidth=1, label=f'Mean of Diffs ({np.mean(curr["m_diff"]):.2f})')
        
        ax_diff.set_title("Mean Difference (M₁ - M₂)")
        ax_diff.set_xlabel("Mean Difference")
        ax_diff.legend(loc='upper right')

        # # --- Calculate Global Bins for Bottom Panels ---
        # # Ensures all histograms in the same panel have identical bin widths
        # ss_min = min(np.min(curr['ss1']), np.min(curr['ss2']), np.min(curr['diff_ss']), np.min(curr['ss_grand']))
        # ss_max = max(np.max(curr['ss1']), np.max(curr['ss2']), np.max(curr['diff_ss']), np.max(curr['ss_grand']))
        # ss_bins = np.linspace(ss_min, ss_max, 100)

        # ms_min = min(np.nanmin(curr['var1']), np.nanmin(curr['var2']), np.nanmin(curr['diff_var']), np.nanmin(curr['var_grand']))
        # ms_max = max(np.nanmax(curr['var1']), np.nanmax(curr['var2']), np.nanmax(curr['diff_var']), np.nanmax(curr['var_grand']))
        # ms_bins = np.linspace(ms_min, ms_max, 100)

        # --- BOTTOM LEFT: Sampling Distribution of SS ---
        ax_ss.hist(curr['ss1'], bins=30, color='purple', alpha=0.5, label='$SS_1$')
        ax_ss.hist(curr['ss2'], bins=30, color='darkorange', alpha=0.5, label='$SS_2$')
        ax_ss.hist(curr['diff_ss'], bins=30, color='seagreen', alpha=0.6, label='$SS_{between}$')
        ax_ss.hist(curr['ss_grand'], bins=30, color='dodgerblue', alpha=0.6, label='$SS_{grand}$')
        
        ax_ss.axvline(np.mean(curr['ss1']), color='purple', linestyle='dashed', linewidth=1)
        ax_ss.axvline(np.mean(curr['ss2']), color='darkorange', linestyle='dashed', linewidth=1)
        ax_ss.axvline(np.mean(curr['diff_ss']), color='seagreen', linestyle='dashed', linewidth=1)
        ax_ss.axvline(np.mean(curr['ss_grand']), color='dodgerblue', linestyle='dashed', linewidth=1)
        
        ax_ss.set_title("Sampling Dist. of Sum of Squares (SS)")
        ax_ss.set_xlabel("Sum of Squares")
        ax_ss.legend(loc='upper right')

        # --- BOTTOM RIGHT: Sampling Distribution of Mean Squares (MS) ---
        ax_ms.hist(curr['var1'], bins=30, color='purple', alpha=0.5, label='$MS_1$ (s²₁)')
        ax_ms.hist(curr['var2'], bins=30, color='darkorange', alpha=0.5, label='$MS_2$ (s²₂)')
        ax_ms.hist(curr['diff_var'], bins=100, color='seagreen', alpha=0.6, label='$MS_{between}$')
        ax_ms.hist(curr['var_grand'], bins=100, color='dodgerblue', alpha=0.6, label='$MS_{grand}$')
        
        ax_ms.axvline(true_var, color='black', linestyle='--', linewidth=1, label=f'True σ² ({true_var:.3f})')
        
        ax_ms.axvline(np.nanmean(curr['var1']), color='purple', linestyle='dashed', linewidth=1)
        ax_ms.axvline(np.nanmean(curr['var2']), color='darkorange', linestyle='dashed', linewidth=1)
        ax_ms.axvline(np.mean(curr['diff_var']), color='seagreen', linestyle='dashed', linewidth=1)
        ax_ms.axvline(np.mean(curr['var_grand']), color='dodgerblue', linestyle='dashed', linewidth=1)
        
        ax_ms.set_title("Sampling Dist. of Mean Squares (MS)")
        ax_ms.set_xlabel("Variance Estimate (MS)")
        ax_ms.legend(loc='upper right')

        return fig

app = App(app_ui, server)
